# Pilas con listas enlazadas simples
## CÃ³digo completo
```c
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>

struct nodo{
    int info;
    struct nodo *sig;
};
typedef struct nodo nodo_t;

struct pila{
    nodo_t *tope;
    int tam;
};
typedef struct pila pila_t;

void inicializar(pila_t *);
nodo_t *crea_nodo();
bool esta_vacia(const pila_t *);
bool esta_llena(const pila_t *);
void apilar(pila_t *, int);
int desapilar(pila_t *);
int tam(const pila_t *);
void liberar(pila_t *);

int main(){
    pila_t pila;
    inicializar(&pila);
    apilar(&pila, 1);
    apilar(&pila, 2);
    printf("%d\n", desapilar(&pila)); //output: 2
    printf("%d\n", desapilar(&pila)); //output: 1
    liberar(&pila);
    return 0;
}

void inicializar(pila_t *pila){
    pila->tope = NULL;
    pila->tam = 0;
}

nodo_t *crea_nodo(){
    nodo_t *nodo = NULL;
    nodo = (nodo_t *)malloc(sizeof(nodo_t));
    assert(nodo != NULL);
    nodo->info = 0;
    nodo->sig = NULL;
    return nodo;
}

bool esta_vacia(const pila_t *p){
    if(p->tam == 0){
        return true;
    }
    else{
        return false;
    }
}

bool esta_llena(const pila_t *p){
    return false;
}

void apilar(pila_t *p, int dato){
    nodo_t *nodo = crea_nodo();
    nodo->sig = p->tope;
    nodo->info = dato;
    p->tope = nodo;
    (p->tam)++;
}

int desapilar(pila_t *p){
    nodo_t *tmp = p->tope;
    if(p->tam > 0){
        p->tope = p->tope->sig;
        int dato = tmp->info;
        free(tmp);
        (p->tam)--;
        return dato;
    }
}

int tam(const pila_t *p){
    return p->tam;
}

void liberar(pila_t *p){
    for(nodo_t *it = p->tope; it != NULL; it = it->sig){
        nodo_t *tmp = it;
        it = it->sig;
        free(tmp);
    }
}
```
