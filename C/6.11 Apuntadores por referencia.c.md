# Apuntadores por referencia
```c
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

void crea_arreglo(int **, int);
int main(){
    int *arreglo = NULL;
    crea_arreglo(&arreglo, 10);
    for(int i = 0; i < 10; i++){
        scanf("%d", arreglo + i);
    }
    for(int i = 0; i < 10; i++){
        printf("%d", *(arreglo + i));
    }
    free(arreglo);
    return 0;
}

void crea_arreglo(int **arreglo, int tam){
    *arreglo = (int *)malloc(tam * sizeof(int));
    assert(*arreglo != NULL);
}
```
