# Colas con listas enlazadas simples
## CÃ³digo completo
```c
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <stdbool.h>

struct nodo{
    int info;
    struct nodo *sig;
};
typedef struct nodo nodo_t;

typedef struct{
    nodo_t *frente;
    nodo_t *atras;
    int tam;
}cola_t;

void inicializa_cola(cola_t *);
nodo_t *crea_nodo();
void encolar(cola_t *, int);
int desencolar(cola_t *);

int main(){
    cola_t cola;
    inicializa_cola(&cola);
    printf("%d\n", desencolar(&cola)); //output: -1
    encolar(&cola, 5);
    encolar(&cola, 8);
    printf("%d\n", desencolar(&cola)); //output: 5
    printf("%d\n", desencolar(&cola)); //output: 8
    printf("%d\n", desencolar(&cola)); //output: -1
    return 0;
}

void inicializa_cola(cola_t *c){
    c->frente = NULL;
    c->atras = NULL;
    c->tam = 0;
}

nodo_t *crea_nodo(){
    nodo_t *nodo = NULL;
    nodo = (nodo_t *)malloc(sizeof(nodo_t));
    assert(nodo != NULL);
    nodo->info = 0;
    nodo->sig = NULL;
    return nodo;
}

void encolar(cola_t *c, int dato){
    nodo_t *nodo = crea_nodo();
    nodo->info = dato;
    nodo->sig = NULL;
    if(c->atras != NULL){
        c->atras->sig = nodo;
        c->atras = nodo;
    }
    else{
        c->atras = c->frente = nodo;
    }
    c->tam++;
}

int desencolar(cola_t *c){
    if(c->frente != NULL){
        int dato = c->frente->info;
        nodo_t *tmp = c->frente;
        c->frente = c->frente->sig;
        free(tmp);
        if(c->frente == NULL){
            c->atras = NULL;
        }
        c->tam--;
        return dato;
    }
    else{
        return -1;
    }
}
```
