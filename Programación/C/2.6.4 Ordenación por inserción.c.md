# Ordenación por inserción
```
void ordena_insercion(int *a, int n){
	for(int i = 1; i < n; i++){
		int j = i;
		while(a[j] < a[j-1] && j > 0){
			swap(&a[j], &a[j-1]);
			j--;
		}
	}
}
```

```
void ordena_insercion(int *a, int n){
	for(int i = 1; i < n; i++){
		int j = i;
		while(a[j] < a[j-1] && j > 0){
			swap(&a[j], &a[j-1]);
			j--;
		}
	}
}

void swap(int *a, int *b){
	int t = *a;
	*a = *b;
	*b = t;
}
```
